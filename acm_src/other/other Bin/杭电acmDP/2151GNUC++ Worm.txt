#include"iostream"
using namespace std;
int main()
{
    int p,m,n,t,i,b[101],a[101];
    while(cin>>n>>p>>m>>t)
    {
        memset(a,0,sizeof(a));
        a[p]=1;
        while(m--)
        {
            for(i=1;i<=n;i++) //第i位置由i-1和i＋1位置过来，所以种类都复给i
                b[i]=a[i-1]+a[i+1];
            for(i=1;i<=n;i++)  //每分钟保存一个状态
                a[i]=b[i];
        }
        cout<<b[t]<<endl;
    }
    return 0;
}
