#include"stdio.h"
int q[4],m[13],dp[14][700000];
int main()
{
    int n,i,j,k,t,max,min;
    while(scanf("%d",&n)!=EOF&&n)
    {
        max=-1;
        scanf("%d %d %d",&q[1],&q[2],&q[3]);
        for(i=1;i<=n;i++)
        {
            scanf("%d",&m[i]);
            if(m[i]>max)
                max=m[i];
        }
        for(i=m[1];i<=max;i++)
            dp[1][i]=i*q[1]+i*q[2];
        for(i=2;i<=n;i++)
        {
            for(j=m[i];j<=max;j++)
            {
                min=1000000;
                for(k=m[i-1];k<=max;k++)
                {
                    if(j>=k)
                        t=dp[i-1][k]+q[1]*(j-k)+q[2]*j;
                    else
                        t=dp[i-1][k]+(k-j)*q[3]+q[2]*j;
                    if(t<min)
                        min=t;
                }
                dp[i][j]=min;
            }
        }
        min=1000000;
        for(j=m[n];j<=max;j++)
        {
            if(dp[n][j]<min)
                min=dp[n][j];
        }
        printf("%d\n",min);
    }
    return 0;
}
