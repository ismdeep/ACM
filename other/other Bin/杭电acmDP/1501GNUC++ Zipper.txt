#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char str1[100000], str2[100000], str3[100000];
char hash[200][200];

int DFS(int a, int b, int c)
{
    if (str3[c] == '\0') return 1;
    if (hash[a][b]) return 0;
    hash[a][b] = 1;
    if (str1[a] == str3[c] && DFS(a + 1, b, c + 1)) return 1;
    if (str2[b] == str3[c] && DFS(a, b + 1, c + 1)) return 1;
    return 0;
}

int main(void)
{
    int i, T;
    scanf("%d", &T);
    for (i = 1; i <= T; i++)
    {
        memset(hash, 0, sizeof(hash));
        scanf("%s%s%s", str1, str2, str3);
        printf("Data set %d: ", i);
        puts(DFS(0, 0, 0) ? "yes" : "no");
    }
    return 0;
}
