/* 1466 递归  nice!
平面上有n条直线，且无三线共点，问这些直线能有多少种不同交点数
如果有n条线
有以下情况：
1条平行（全不平行）
（n-1）条线的所有情况+(n-1)*1
2条平行（n-2）条线的所有情况+(n-2)*2
...
i条平行（n-2）条线的所有情况+(n-i)*i
...
n-1条平行1条线的所有情况+ 1*(n-1)
*/
#include<iostream>
using namespace std;
int main()
{//f[j]表示i条边时，是否能产生j个结点，能，返回1，不能，返回0
	int i,j,n,f[21][191];
	for(i=0;i<21;i++)
		for(j=0;j<191;j++)
			f[i][j]=(j==0);//f[0]都置1
	for(n=2;n<21;n++)
		for(i=n-1;i>=1;i--)
			for(j=0;j<191;j++)
				if(f[n-i][j]==1)
					f[n][j+(n-i)*i]=1;
	//如果f[n][x]==1
	//f[n][x+((n-i)*i)]=1
	while(cin>>n)
	{
		cout<<"0";
		for(j=1;j<=n*(n-1)/2;j++)
			if(f[n][j])
				cout<<" "<<j;
		cout<<endl;
	}
}
